# scripts/merge_audio.py 사용 예시

# 1. 기본 사용법: 'data/03_tts_output/' 디렉토리의 모든 '.wav' 파일을 병합하여 'merged_output.wav'로 저장합니다. (기본 묵음 200ms)
venv/bin/python3 scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_output.wav

# 2. 특정 Reference Audio의 파일만 병합: 'my_voice' 참조 오디오로 생성된 파일들만 병합하여 'merged_my_voice.wav'로 저장합니다.
#    패턴: '*_my_voice_*.wav' (예: video_title_my_voice_2023_11_30_1.wav)
venv/bin/python3 scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_my_voice.wav --pattern "START-370-*.wav"

# 3. 묵음 간격 조절: 파일 사이에 500ms의 묵음을 추가하여 병합합니다.
venv/bin/python3 scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_with_500ms_silence.wav --silence_ms 500

# 4. 특정 Reference Audio와 묵음 간격 조절을 함께 사용:
venv/bin/python3 scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_another_voice_500ms.wav --pattern "*_another_voice_*.wav" --silence_ms 500

# 5. 다른 오디오 파일 형식 병합 (예: .mp3 파일):
#    만약 디렉토리에 .mp3 파일이 있다면, 패턴을 변경하여 병합할 수 있습니다.
#    (merge_audio.py 스크립트는 .wav, .mp3, .flac 형식을 지원합니다.)
# python scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_mp3_output.mp3 --pattern "*.mp3"

# 6. 특정 비디오 파일 이름으로 시작하는 파일만 병합:
#    예를 들어, 'my_video'라는 비디오에서 생성된 파일들만 병합하고 싶을 때.
python scripts/merge_audio.py data/03_tts_output/ data/03_tts_output/merged_my_video.wav --pattern "my_video_*.wav"
